Предложение:
	НГ+НГ
	НГ+ИГ
	ГГ+ГГ
	ИГ+ИГ
	ГГ+ИГ ("наступило" + "лето", род, им.падеж, число (одуш.ед. + мн., если в ИГ есть др. одуш. ИГ))
	ИГ+ГГ (-//-)
Именные:
	Существительное +:
		прил+СУЩ (им.падеж, род, число)
		прич+СУЩ
		мест+СУЩ
		числ+СУЩ
		СУЩ+сущ
		СУЩ+мест
		СУЩ+нареч
		СУЩ+комп
		предл+СУЩ
	Прилагательное +:
		нареч+ПРИЛ
		ПРИЛ+мест
		ПРИЛ+инф
		ПРИЛ+сущ
		предл+ПРИл (если нет СУЩ и МЕСТ и ПРИЧ)
	Числительное +:
		ЧИСЛ+сущ
		предл+ЧИСЛ (если нет СУЩ и МЕСТ и ПРИЧ и ПРИЛ)
	Местоимение +:
		МЕСТ+прил
		МЕСТ+сущ
		предл+МЕСТ (если нет СУЩ)
Глагольные:
	Инфинитив +:
		ИНФ+сущ
		ИНФ+деепр
		нареч+ИНФ
		ИНФ+инф
	Глагол +:
		ГЛ+сущ ("насорил"+ "на пляже", ГГ+ИГ (гл+прямое дополнение), все падежи ИГ, кроме ИМ.п.)
		ГЛ+инф
		ГЛ+деепр
		нареч+ГЛ ("сегодня" + "лил")
	Причастие +:
		ПРИЧ+мест
		предл+ПРИЧ (если нет СУЩ и МЕСТ)
	Деепричастие +:
		ДЕЕПР+сущ
Наречные:
	Наречие +:
		НАРЕЧ+сущ ("ближе" + "к краю", НГ+ИГ, все падежи ИГ, кроме ИМ.п.)
		нареч+НАРЕЧ
